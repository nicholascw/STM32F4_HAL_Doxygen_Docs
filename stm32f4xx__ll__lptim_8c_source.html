140 ErrorStatus LL_LPTIM_Init(LPTIM_TypeDef *LPTIMx, LL_LPTIM_InitTypeDef *LPTIM_InitStruct)
141 {
142  ErrorStatus result = SUCCESS;
143  /* Check the parameters */
144  assert_param(IS_LPTIM_INSTANCE(LPTIMx));
145  assert_param(IS_LL_LPTIM_CLOCK_SOURCE(LPTIM_InitStruct->ClockSource));
146  assert_param(IS_LL_LPTIM_CLOCK_PRESCALER(LPTIM_InitStruct->Prescaler));
147  assert_param(IS_LL_LPTIM_WAVEFORM(LPTIM_InitStruct->Waveform));
148  assert_param(IS_LL_LPTIM_OUTPUT_POLARITY(LPTIM_InitStruct->Polarity));
149 
150  /* The LPTIMx_CFGR register must only be modified when the LPTIM is disabled
151  (ENABLE bit is reset to 0).
152  */
153  if (LL_LPTIM_IsEnabled(LPTIMx) == 1UL)
154  {
155  result = ERROR;
156  }
157  else
158  {
159  /* Set CKSEL bitfield according to ClockSource value */
160  /* Set PRESC bitfield according to Prescaler value */
161  /* Set WAVE bitfield according to Waveform value */
162  /* Set WAVEPOL bitfield according to Polarity value */
163  MODIFY_REG(LPTIMx->CFGR,
164  (LPTIM_CFGR_CKSEL | LPTIM_CFGR_PRESC | LPTIM_CFGR_WAVE | LPTIM_CFGR_WAVPOL),
165  LPTIM_InitStruct->ClockSource | \
166  LPTIM_InitStruct->Prescaler | \
167  LPTIM_InitStruct->Waveform | \
168  LPTIM_InitStruct->Polarity);
169  }
170 
171  return result;
172 }

191 void LL_LPTIM_Disable(LPTIM_TypeDef *LPTIMx)
192 {
193  LL_RCC_ClocksTypeDef rcc_clock;
194  uint32_t tmpclksource = 0;
195  uint32_t tmpIER;
196  uint32_t tmpCFGR;
197  uint32_t tmpCMP;
198  uint32_t tmpARR;
199  uint32_t tmpOR;
200 
201  /* Check the parameters */
202  assert_param(IS_LPTIM_INSTANCE(LPTIMx));
203 
204  __disable_irq();
205 
206  /********** Save LPTIM Config *********/
207  /* Save LPTIM source clock */
208  switch ((uint32_t)LPTIMx)
209  {
210  case LPTIM1_BASE:
211  tmpclksource = LL_RCC_GetLPTIMClockSource(LL_RCC_LPTIM1_CLKSOURCE);
212  break;
213  default:
214  break;
215  }
216 
217  /* Save LPTIM configuration registers */
218  tmpIER = LPTIMx->IER;
219  tmpCFGR = LPTIMx->CFGR;
220  tmpCMP = LPTIMx->CMP;
221  tmpARR = LPTIMx->ARR;
222  tmpOR = LPTIMx->OR;
223 
224  /************* Reset LPTIM ************/
225  (void)LL_LPTIM_DeInit(LPTIMx);
226 
227  /********* Restore LPTIM Config *******/
228  LL_RCC_GetSystemClocksFreq(&rcc_clock);
229 
230  if ((tmpCMP != 0UL) || (tmpARR != 0UL))
231  {
232  /* Force LPTIM source kernel clock from APB */
233  switch ((uint32_t)LPTIMx)
234  {
235  case LPTIM1_BASE:
236  LL_RCC_SetLPTIMClockSource(LL_RCC_LPTIM1_CLKSOURCE_PCLK1);
237  break;
238  default:
239  break;
240  }
241 
242  if (tmpCMP != 0UL)
243  {
244  /* Restore CMP and ARR registers (LPTIM should be enabled first) */
245  LPTIMx->CR |= LPTIM_CR_ENABLE;
246  LPTIMx->CMP = tmpCMP;
247 
248  /* Polling on CMP write ok status after above restore operation */
249  do
250  {
251  rcc_clock.SYSCLK_Frequency--; /* Used for timeout */
252  } while (((LL_LPTIM_IsActiveFlag_CMPOK(LPTIMx) != 1UL)) && ((rcc_clock.SYSCLK_Frequency) > 0UL));
253 
254  LL_LPTIM_ClearFlag_CMPOK(LPTIMx);
255  }
256 
257  if (tmpARR != 0UL)
258  {
259  LPTIMx->CR |= LPTIM_CR_ENABLE;
260  LPTIMx->ARR = tmpARR;
261 
262  LL_RCC_GetSystemClocksFreq(&rcc_clock);
263  /* Polling on ARR write ok status after above restore operation */
264  do
265  {
266  rcc_clock.SYSCLK_Frequency--; /* Used for timeout */
267  } while (((LL_LPTIM_IsActiveFlag_ARROK(LPTIMx) != 1UL)) && ((rcc_clock.SYSCLK_Frequency) > 0UL));
268 
269  LL_LPTIM_ClearFlag_ARROK(LPTIMx);
270  }
271 
272  /* Restore LPTIM source kernel clock */
273  LL_RCC_SetLPTIMClockSource(tmpclksource);
274  }
275 
276  /* Restore configuration registers (LPTIM should be disabled first) */
277  LPTIMx->CR &= ~(LPTIM_CR_ENABLE);
278  LPTIMx->IER = tmpIER;
279  LPTIMx->CFGR = tmpCFGR;
280  LPTIMx->OR = tmpOR;
281 
282  __enable_irq();
283 }
