665 void HAL_I2SEx_FullDuplex_IRQHandler(I2S_HandleTypeDef *hi2s)
666 {
667  __IO uint32_t i2ssr = hi2s->Instance->SR;
668  __IO uint32_t i2sextsr = I2SxEXT(hi2s->Instance)->SR;
669  __IO uint32_t i2scr2 = hi2s->Instance->CR2;
670  __IO uint32_t i2sextcr2 = I2SxEXT(hi2s->Instance)->CR2;
671 
672  /* Check if the I2S_MODE_MASTER_TX or I2S_MODE_SLAVE_TX Mode is selected */
673  if ((hi2s->Init.Mode == I2S_MODE_MASTER_TX) || (hi2s->Init.Mode == I2S_MODE_SLAVE_TX))
674  {
675  /* I2S in mode Transmitter -------------------------------------------------*/
676  if (((i2ssr & I2S_FLAG_TXE) == I2S_FLAG_TXE) && ((i2scr2 & I2S_IT_TXE) != RESET))
677  {
678  /* When the I2S mode is configured as I2S_MODE_MASTER_TX or I2S_MODE_SLAVE_TX,
679  the I2S TXE interrupt will be generated to manage the full-duplex transmit phase. */
680  I2SEx_TxISR_I2S(hi2s);
681  }
682 
683  /* I2Sext in mode Receiver -----------------------------------------------*/
684  if (((i2sextsr & I2S_FLAG_RXNE) == I2S_FLAG_RXNE) && ((i2sextcr2 & I2S_IT_RXNE) != RESET))
685  {
686  /* When the I2S mode is configured as I2S_MODE_MASTER_TX or I2S_MODE_SLAVE_TX,
687  the I2Sext RXNE interrupt will be generated to manage the full-duplex receive phase. */
688  I2SEx_RxISR_I2SExt(hi2s);
689  }
690 
691  /* I2Sext Overrun error interrupt occurred --------------------------------*/
692  if (((i2sextsr & I2S_FLAG_OVR) == I2S_FLAG_OVR) && ((i2sextcr2 & I2S_IT_ERR) != RESET))
693  {
694  /* Disable RXNE and ERR interrupt */
695  __HAL_I2SEXT_DISABLE_IT(hi2s, (I2S_IT_RXNE | I2S_IT_ERR));
696 
697  /* Disable TXE and ERR interrupt */
698  __HAL_I2S_DISABLE_IT(hi2s, (I2S_IT_TXE | I2S_IT_ERR));
699 
700  /* Clear Overrun flag */
701  __HAL_I2S_CLEAR_OVRFLAG(hi2s);
702 
703  /* Set the I2S State ready */
704  hi2s->State = HAL_I2S_STATE_READY;
705 
706  /* Set the error code and execute error callback*/
707  SET_BIT(hi2s->ErrorCode, HAL_I2S_ERROR_OVR);
708  /* Call user error callback */
709 #if (USE_HAL_I2S_REGISTER_CALLBACKS == 1U)
710  hi2s->ErrorCallback(hi2s);
711 #else
712  HAL_I2S_ErrorCallback(hi2s);
713 #endif /* USE_HAL_I2S_REGISTER_CALLBACKS */
714  }
715 
716  /* I2S Underrun error interrupt occurred ----------------------------------*/
717  if (((i2ssr & I2S_FLAG_UDR) == I2S_FLAG_UDR) && ((i2scr2 & I2S_IT_ERR) != RESET))
718  {
719  /* Disable TXE and ERR interrupt */
720  __HAL_I2S_DISABLE_IT(hi2s, (I2S_IT_TXE | I2S_IT_ERR));
721 
722  /* Disable RXNE and ERR interrupt */
723  __HAL_I2SEXT_DISABLE_IT(hi2s, (I2S_IT_RXNE | I2S_IT_ERR));
724 
725  /* Clear underrun flag */

729  hi2s->State = HAL_I2S_STATE_READY;
