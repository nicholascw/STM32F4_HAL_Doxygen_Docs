365 HAL_StatusTypeDef HAL_SAI_Init(SAI_HandleTypeDef *hsai)
366 {
367  uint32_t tmpregisterGCR = 0U;
368 
369  /* This variable used to store the SAI_CK_x (value in Hz) */
370  uint32_t freq = 0U;
371 
372  /* This variable is used to compute CKSTR bits of SAI CR1 according to
373  ClockStrobing and AudioMode fields */
374  uint32_t ckstr_bits = 0U;
375  uint32_t syncen_bits = 0U;
376 
377  /* Check the SAI handle allocation */
378  if(hsai == NULL)
379  {
380  return HAL_ERROR;
381  }
382 
383  /* check the instance */
384  assert_param(IS_SAI_ALL_INSTANCE(hsai->Instance));
385 
386  /* Check the SAI Block parameters */
387  assert_param(IS_SAI_AUDIO_FREQUENCY(hsai->Init.AudioFrequency));
388  assert_param(IS_SAI_BLOCK_PROTOCOL(hsai->Init.Protocol));
389  assert_param(IS_SAI_BLOCK_MODE(hsai->Init.AudioMode));
390  assert_param(IS_SAI_BLOCK_SYNCEXT(hsai->Init.SynchroExt));
391  assert_param(IS_SAI_BLOCK_DATASIZE(hsai->Init.DataSize));
392  assert_param(IS_SAI_BLOCK_FIRST_BIT(hsai->Init.FirstBit));
393  assert_param(IS_SAI_BLOCK_CLOCK_STROBING(hsai->Init.ClockStrobing));
394  assert_param(IS_SAI_BLOCK_SYNCHRO(hsai->Init.Synchro));
395  assert_param(IS_SAI_BLOCK_OUTPUT_DRIVE(hsai->Init.OutputDrive));
396  assert_param(IS_SAI_BLOCK_NODIVIDER(hsai->Init.NoDivider));
397  assert_param(IS_SAI_BLOCK_FIFO_THRESHOLD(hsai->Init.FIFOThreshold));
398  assert_param(IS_SAI_MONO_STEREO_MODE(hsai->Init.MonoStereoMode));
399  assert_param(IS_SAI_BLOCK_COMPANDING_MODE(hsai->Init.CompandingMode));
400  assert_param(IS_SAI_BLOCK_TRISTATE_MANAGEMENT(hsai->Init.TriState));
401 
402  /* Check the SAI Block Frame parameters */
403  assert_param(IS_SAI_BLOCK_FRAME_LENGTH(hsai->FrameInit.FrameLength));
404  assert_param(IS_SAI_BLOCK_ACTIVE_FRAME(hsai->FrameInit.ActiveFrameLength));
405  assert_param(IS_SAI_BLOCK_FS_DEFINITION(hsai->FrameInit.FSDefinition));
406  assert_param(IS_SAI_BLOCK_FS_POLARITY(hsai->FrameInit.FSPolarity));
407  assert_param(IS_SAI_BLOCK_FS_OFFSET(hsai->FrameInit.FSOffset));
408 
409  /* Check the SAI Block Slot parameters */
410  assert_param(IS_SAI_BLOCK_FIRSTBIT_OFFSET(hsai->SlotInit.FirstBitOffset));
411  assert_param(IS_SAI_BLOCK_SLOT_SIZE(hsai->SlotInit.SlotSize));
412  assert_param(IS_SAI_BLOCK_SLOT_NUMBER(hsai->SlotInit.SlotNumber));
413  assert_param(IS_SAI_SLOT_ACTIVE(hsai->SlotInit.SlotActive));
414 
415  if(hsai->State == HAL_SAI_STATE_RESET)
416  {
417  /* Allocate lock resource and initialize it */
418  hsai->Lock = HAL_UNLOCKED;
419 
420 #if (USE_HAL_SAI_REGISTER_CALLBACKS == 1)
421  /* Reset callback pointers to the weak predefined callbacks */
422  hsai->RxCpltCallback = HAL_SAI_RxCpltCallback;
423  hsai->RxHalfCpltCallback = HAL_SAI_RxHalfCpltCallback;
424  hsai->TxCpltCallback = HAL_SAI_TxCpltCallback;
425  hsai->TxHalfCpltCallback = HAL_SAI_TxHalfCpltCallback;
426  hsai->ErrorCallback = HAL_SAI_ErrorCallback;
427 
428  /* Init the low level hardware : GPIO, CLOCK, NVIC and DMA */
429  if (hsai->MspInitCallback == NULL)
430  {
431  hsai->MspInitCallback = HAL_SAI_MspInit;
432  }
433  hsai->MspInitCallback(hsai);
434 #else
435  /* Init the low level hardware : GPIO, CLOCK, NVIC and DMA */
436  HAL_SAI_MspInit(hsai);
437 #endif /* USE_HAL_SAI_REGISTER_CALLBACKS */
438  }
439 
440  hsai->State = HAL_SAI_STATE_BUSY;
441 
442  /* Disable the selected SAI peripheral */
443  SAI_Disable(hsai);
444 
445  /* SAI Block Synchro Configuration -----------------------------------------*/
446  SAI_BlockSynchroConfig(hsai);
447 
448  /* Configure Master Clock using the following formula :
449  MCLK_x = SAI_CK_x / (MCKDIV[3:0] * 2) with MCLK_x = 256 * FS
450  FS = SAI_CK_x / (MCKDIV[3:0] * 2) * 256
451  MCKDIV[3:0] = SAI_CK_x / FS * 512 */
452  if(hsai->Init.AudioFrequency != SAI_AUDIO_FREQUENCY_MCKDIV)
453  {
454  /* Get SAI clock source based on Source clock selection from RCC */
455  freq = SAI_GetInputClock(hsai);
456 
457  /* (saiclocksource x 10) to keep Significant digits */
458  tmpregisterGCR = (((freq * 10U) / ((hsai->Init.AudioFrequency) * 512U)));
459 
460  hsai->Init.Mckdiv = tmpregisterGCR / 10U;
461 
462  /* Round result to the nearest integer */
463  if((tmpregisterGCR % 10U) > 8U)
464  {
465  hsai->Init.Mckdiv+= 1U;
466  }
467  }
468 
469  /* Compute CKSTR bits of SAI CR1 according to ClockStrobing and AudioMode */
470  if((hsai->Init.AudioMode == SAI_MODEMASTER_TX) || (hsai->Init.AudioMode == SAI_MODESLAVE_TX))
471  {
472  ckstr_bits = (hsai->Init.ClockStrobing == SAI_CLOCKSTROBING_RISINGEDGE) ? 0U: SAI_xCR1_CKSTR;
473  }
474  else
475  {
476  ckstr_bits = (hsai->Init.ClockStrobing == SAI_CLOCKSTROBING_RISINGEDGE) ? SAI_xCR1_CKSTR: 0U;
477  }
478 
479  /* SAI Block Configuration -------------------------------------------------*/
480  switch(hsai->Init.Synchro)
481  {
482  case SAI_ASYNCHRONOUS :
483  {
484  syncen_bits = 0U;
485  }
486  break;
487  case SAI_SYNCHRONOUS :
488  {
489  syncen_bits = SAI_xCR1_SYNCEN_0;
490  }
491  break;
492  case SAI_SYNCHRONOUS_EXT_SAI1 :
493  case SAI_SYNCHRONOUS_EXT_SAI2 :
494  {
495  syncen_bits = SAI_xCR1_SYNCEN_1;
496  }
497  break;
498  default:
499  break;
500  }
501  /* SAI CR1 Configuration */
502  hsai->Instance->CR1 &= ~(SAI_xCR1_MODE | SAI_xCR1_PRTCFG | SAI_xCR1_DS | \
503  SAI_xCR1_LSBFIRST | SAI_xCR1_CKSTR | SAI_xCR1_SYNCEN |\
504  SAI_xCR1_MONO | SAI_xCR1_OUTDRIV | SAI_xCR1_DMAEN | \
505  SAI_xCR1_NODIV | SAI_xCR1_MCKDIV);
506 
507  hsai->Instance->CR1 |= (hsai->Init.AudioMode | hsai->Init.Protocol | \
508  hsai->Init.DataSize | hsai->Init.FirstBit | \
509  ckstr_bits | syncen_bits | \
510  hsai->Init.MonoStereoMode | hsai->Init.OutputDrive | \
511  hsai->Init.NoDivider | (hsai->Init.Mckdiv << 20U));
512 
513  /* SAI CR2 Configuration */
514  hsai->Instance->CR2 &= ~(SAI_xCR2_FTH | SAI_xCR2_FFLUSH | SAI_xCR2_COMP | SAI_xCR2_CPL);
515  hsai->Instance->CR2 |= (hsai->Init.FIFOThreshold | hsai->Init.CompandingMode | hsai->Init.TriState);
516 
517  /* SAI Frame Configuration -----------------------------------------*/
518  hsai->Instance->FRCR&=(~(SAI_xFRCR_FRL | SAI_xFRCR_FSALL | SAI_xFRCR_FSDEF | \
519  SAI_xFRCR_FSPOL | SAI_xFRCR_FSOFF));
520  hsai->Instance->FRCR|=((hsai->FrameInit.FrameLength - 1U) |
521  hsai->FrameInit.FSOffset |
522  hsai->FrameInit.FSDefinition |
523  hsai->FrameInit.FSPolarity |
524  ((hsai->FrameInit.ActiveFrameLength - 1U) << 8U));
525 
526  /* SAI Block_x SLOT Configuration ------------------------------------------*/
527  /* This register has no meaning in AC 97 and SPDIF audio protocol */
528  hsai->Instance->SLOTR &= ~(SAI_xSLOTR_FBOFF | SAI_xSLOTR_SLOTSZ | \
529  SAI_xSLOTR_NBSLOT | SAI_xSLOTR_SLOTEN );
530 
531  hsai->Instance->SLOTR |= hsai->SlotInit.FirstBitOffset | hsai->SlotInit.SlotSize | \
532  (hsai->SlotInit.SlotActive << 16U) | ((hsai->SlotInit.SlotNumber - 1U) << 8U);
533 
534  /* Initialize the error code */
535  hsai->ErrorCode = HAL_SAI_ERROR_NONE;
536 
537  /* Initialize the SAI state */
538  hsai->State= HAL_SAI_STATE_READY;
539 
540  /* Release Lock */
541  __HAL_UNLOCK(hsai);
542 
543  return HAL_OK;
544 }

858 HAL_StatusTypeDef HAL_SAI_Transmit(SAI_HandleTypeDef *hsai, uint8_t* pData, uint16_t Size, uint32_t Timeout)
859 {
860  uint32_t tickstart = HAL_GetTick();
861 
862  if((pData == NULL ) || (Size == 0))
863  {
864  return HAL_ERROR;
865  }
866 
867  if(hsai->State == HAL_SAI_STATE_READY)
868  {
869  /* Process Locked */
870  __HAL_LOCK(hsai);
871 
872  hsai->XferSize = Size;
873  hsai->XferCount = Size;
874  hsai->pBuffPtr = pData;
875  hsai->State = HAL_SAI_STATE_BUSY_TX;
876  hsai->ErrorCode = HAL_SAI_ERROR_NONE;
877 
878  /* Check if the SAI is already enabled */
879  if((hsai->Instance->CR1 & SAI_xCR1_SAIEN) == RESET)
880  {
881  /* fill the fifo with data before to enabled the SAI */
882  SAI_FillFifo(hsai);
883  /* Enable SAI peripheral */
884  __HAL_SAI_ENABLE(hsai);
885  }
886 
887  while(hsai->XferCount > 0U)
888  {
889  /* Write data if the FIFO is not full */
890  if((hsai->Instance->SR & SAI_xSR_FLVL) != SAI_FIFOSTATUS_FULL)
891  {
892  if((hsai->Init.DataSize == SAI_DATASIZE_8) && (hsai->Init.CompandingMode == SAI_NOCOMPANDING))
893  {
894  hsai->Instance->DR = (*hsai->pBuffPtr++);
895  }
896  else if(hsai->Init.DataSize <= SAI_DATASIZE_16)
897  {
898  hsai->Instance->DR = *((uint16_t *)hsai->pBuffPtr);
899  hsai->pBuffPtr+= 2U;
900  }
901  else
902  {
903  hsai->Instance->DR = *((uint32_t *)hsai->pBuffPtr);
904  hsai->pBuffPtr+= 4U;
905  }
906  hsai->XferCount--;
907  }
908  else
909  {
910  /* Check for the Timeout */
911  if((Timeout != HAL_MAX_DELAY) && ((Timeout == 0U)||((HAL_GetTick() - tickstart) > Timeout)))
912  {
913  /* Update error code */
914  hsai->ErrorCode |= HAL_SAI_ERROR_TIMEOUT;
915 
916  /* Clear all the flags */
917  hsai->Instance->CLRFR = 0xFFFFFFFFU;
918 
919  /* Disable SAI peripheral */
920  SAI_Disable(hsai);
921 
922  /* Flush the fifo */
923  SET_BIT(hsai->Instance->CR2, SAI_xCR2_FFLUSH);
924 
925  /* Change the SAI state */
926  hsai->State = HAL_SAI_STATE_READY;
927 
928  /* Process Unlocked */
929  __HAL_UNLOCK(hsai);
930 
931  return HAL_ERROR;
932  }
933  }
934  }
935 
936  hsai->State = HAL_SAI_STATE_READY;
937 
938  /* Process Unlocked */
939  __HAL_UNLOCK(hsai);
940 
941  return HAL_OK;
942  }
943  else
944  {
945  return HAL_BUSY;
946  }
947 }

958 HAL_StatusTypeDef HAL_SAI_Receive(SAI_HandleTypeDef *hsai, uint8_t *pData, uint16_t Size, uint32_t Timeout)
959 {
960  uint32_t tickstart = HAL_GetTick();
961 
962  if((pData == NULL ) || (Size == 0))
963  {
964  return HAL_ERROR;
965  }
966 
967  if(hsai->State == HAL_SAI_STATE_READY)
968  {
969  /* Process Locked */
970  __HAL_LOCK(hsai);
971 
972  hsai->pBuffPtr = pData;
973  hsai->XferSize = Size;
974  hsai->XferCount = Size;
975  hsai->State = HAL_SAI_STATE_BUSY_RX;
976  hsai->ErrorCode = HAL_SAI_ERROR_NONE;
977 
978  /* Check if the SAI is already enabled */
979  if((hsai->Instance->CR1 & SAI_xCR1_SAIEN) == RESET)
980  {
981  /* Enable SAI peripheral */
982  __HAL_SAI_ENABLE(hsai);
983  }
984 
985  /* Receive data */
986  while(hsai->XferCount > 0U)
987  {
988  if((hsai->Instance->SR & SAI_xSR_FLVL) != SAI_FIFOSTATUS_EMPTY)
989  {
990  if((hsai->Init.DataSize == SAI_DATASIZE_8) && (hsai->Init.CompandingMode == SAI_NOCOMPANDING))
991  {
992  (*hsai->pBuffPtr++) = hsai->Instance->DR;
993  }
994  else if(hsai->Init.DataSize <= SAI_DATASIZE_16)
995  {
996  *((uint16_t*)hsai->pBuffPtr) = hsai->Instance->DR;
997  hsai->pBuffPtr+= 2U;
998  }
999  else
