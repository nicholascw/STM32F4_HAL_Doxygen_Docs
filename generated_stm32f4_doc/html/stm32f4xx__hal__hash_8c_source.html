1336 {

1338  uint32_t tmp_initial_DMATransferSize_inWords;
1339  uint32_t tmp_words_already_pushed;
1340

1342  {
1343  return HAL_ERROR;
1344  }
1345  else
1346  {
1347 
1348  /* Make sure there is enough time to suspend the processing */
1349  tmp_remaining_DMATransferSize_inWords = ((DMA_Stream_TypeDef *)hhash->hdmain->Instance)->NDTR;
1350 
1351  if (tmp_remaining_DMATransferSize_inWords <= HASH_DMA_SUSPENSION_WORDS_LIMIT)
1352  {
1353  /* No suspension attempted since almost to the end of the transferred data. */
1354  /* Best option for user code is to wrap up low priority message hashing */
1355  return HAL_ERROR;
1356  }
1357 
1358  /* Wait for DMAS to be reset */
1359  if (HASH_WaitOnFlagUntilTimeout(hhash, HASH_FLAG_BUSY, SET, HASH_TIMEOUTVALUE) != HAL_OK)
1360  {
1361  return HAL_TIMEOUT;
1362  }
1363 
1364  if (__HAL_HASH_GET_FLAG(HASH_FLAG_DCIS) != RESET)
1365  {
1366  return HAL_ERROR;
1367  }
1368 
1369  /* Wait for DMAS to be set */
1370  if (HASH_WaitOnFlagUntilTimeout(hhash, HASH_FLAG_BUSY, RESET, HASH_TIMEOUTVALUE) != HAL_OK)
1371  {
1372  return HAL_TIMEOUT;
1373  }
1374

1376  if (HAL_DMA_Abort(hhash->hdmain) ==HAL_OK)
1377  {
1378  /*
1379  Note that the Abort function will
1380  - Clear the transfer error flags
1381  - Unlock
1382  - Set the State
1383  */
1384  }
1385 
1386  /* Clear DMAE bit */
1387  CLEAR_BIT(HASH->CR,HASH_CR_DMAE);
1388 
1389  if (HASH_WaitOnFlagUntilTimeout(hhash, HASH_FLAG_BUSY, SET, HASH_TIMEOUTVALUE) != HAL_OK)
1390  {
1391  return HAL_TIMEOUT;
1392  }
1393 
1394  if (__HAL_HASH_GET_FLAG(HASH_FLAG_DCIS) != RESET)
1395  {
1396  return HAL_ERROR;
1397  }
1398 
1399  /* At this point, DMA interface is disabled and no transfer is on-going */
1400  /* Retrieve from the DMA handle how many words remain to be written */
1401  tmp_remaining_DMATransferSize_inWords = ((DMA_Stream_TypeDef *)hhash->hdmain->Instance)->NDTR;
