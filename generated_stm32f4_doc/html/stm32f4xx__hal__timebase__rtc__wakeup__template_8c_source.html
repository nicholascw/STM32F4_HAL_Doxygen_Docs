102 HAL_StatusTypeDef HAL_InitTick (uint32_t TickPriority)
103 {
104  __IO uint32_t counter = 0U;
105 
106  RCC_OscInitTypeDef RCC_OscInitStruct;
107  RCC_PeriphCLKInitTypeDef PeriphClkInitStruct;
108 
109 #ifdef RTC_CLOCK_SOURCE_LSE
110  /* Configue LSE as RTC clock soucre */
111  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_LSE;
112  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_NONE;
113  RCC_OscInitStruct.LSEState = RCC_LSE_ON;
114  PeriphClkInitStruct.RTCClockSelection = RCC_RTCCLKSOURCE_LSE;
115 #elif defined (RTC_CLOCK_SOURCE_LSI)
116  /* Configue LSI as RTC clock soucre */
117  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_LSI;
118  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_NONE;
119  RCC_OscInitStruct.LSIState = RCC_LSI_ON;
120  PeriphClkInitStruct.RTCClockSelection = RCC_RTCCLKSOURCE_LSI;
121 #elif defined (RTC_CLOCK_SOURCE_HSE)
122  /* Configue HSE as RTC clock soucre */
123  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSE;
124  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_NONE;
125  RCC_OscInitStruct.HSEState = RCC_HSE_ON;
126  /* Ensure that RTC is clocked by 1MHz */
127  PeriphClkInitStruct.RTCClockSelection = RCC_RTCCLKSOURCE_1MHZ;
128 #else
129 #error Please select the RTC Clock source
130 #endif /* RTC_CLOCK_SOURCE_LSE */
131 
132  if(HAL_RCC_OscConfig(&RCC_OscInitStruct) == HAL_OK)
133  {
134  PeriphClkInitStruct.PeriphClockSelection = RCC_PERIPHCLK_RTC;
135  if(HAL_RCCEx_PeriphCLKConfig(&PeriphClkInitStruct) == HAL_OK)
136  {
137  /* Enable RTC Clock */
138  __HAL_RCC_RTC_ENABLE();
139  /* The time base should be 1ms
140  Time base = ((RTC_ASYNCH_PREDIV + 1) * (RTC_SYNCH_PREDIV + 1)) / RTC_CLOCK
141  HSE as RTC clock
142  Time base = ((99 + 1) * (9 + 1)) / 1Mhz
143  = 1ms
144  LSE as RTC clock
145  Time base = ((31 + 1) * (0 + 1)) / 32.768Khz
146  = ~1ms
147  LSI as RTC clock
148  Time base = ((31 + 1) * (0 + 1)) / 32Khz
149  = 1ms
150  */
151  hRTC_Handle.Instance = RTC;
152  hRTC_Handle.Init.HourFormat = RTC_HOURFORMAT_24;
153  hRTC_Handle.Init.AsynchPrediv = RTC_ASYNCH_PREDIV;
154  hRTC_Handle.Init.SynchPrediv = RTC_SYNCH_PREDIV;
155  hRTC_Handle.Init.OutPut = RTC_OUTPUT_DISABLE;
156  hRTC_Handle.Init.OutPutPolarity = RTC_OUTPUT_POLARITY_HIGH;
157  hRTC_Handle.Init.OutPutType = RTC_OUTPUT_TYPE_OPENDRAIN;
158  HAL_RTC_Init(&hRTC_Handle);
159 
160  /* Disable the write protection for RTC registers */
161  __HAL_RTC_WRITEPROTECTION_DISABLE(&hRTC_Handle);
162 
163  /* Disable the Wake-up Timer */
164  __HAL_RTC_WAKEUPTIMER_DISABLE(&hRTC_Handle);
165 
166  /* In case of interrupt mode is used, the interrupt source must disabled */
167  __HAL_RTC_WAKEUPTIMER_DISABLE_IT(&hRTC_Handle,RTC_IT_WUT);
168 
169  /* Wait till RTC WUTWF flag is set */
170  while(__HAL_RTC_WAKEUPTIMER_GET_FLAG(&hRTC_Handle, RTC_FLAG_WUTWF) == RESET)
171  {
172  if(counter++ == (SystemCoreClock /48U))
173  {
174  return HAL_ERROR;
175  }
176  }
177 
178  /* Clear PWR wake up Flag */
179  __HAL_PWR_CLEAR_FLAG(PWR_FLAG_WU);
180 
181  /* Clear RTC Wake Up timer Flag */
182  __HAL_RTC_WAKEUPTIMER_CLEAR_FLAG(&hRTC_Handle, RTC_FLAG_WUTF);
183 
184  /* Configure the Wake-up Timer counter */
185  hRTC_Handle.Instance->WUTR = 0U;
186 
187  /* Clear the Wake-up Timer clock source bits in CR register */
188  hRTC_Handle.Instance->CR &= (uint32_t)~RTC_CR_WUCKSEL;
189 
190  /* Configure the clock source */
191  hRTC_Handle.Instance->CR |= (uint32_t)RTC_WAKEUPCLOCK_CK_SPRE_16BITS;
192 
193  /* RTC WakeUpTimer Interrupt Configuration: EXTI configuration */
194  __HAL_RTC_WAKEUPTIMER_EXTI_ENABLE_IT();
195 
196  __HAL_RTC_WAKEUPTIMER_EXTI_ENABLE_RISING_EDGE();
197 
198  /* Configure the Interrupt in the RTC_CR register */
199  __HAL_RTC_WAKEUPTIMER_ENABLE_IT(&hRTC_Handle,RTC_IT_WUT);
200 
201  /* Enable the Wake-up Timer */
202  __HAL_RTC_WAKEUPTIMER_ENABLE(&hRTC_Handle);
203 
204  /* Enable the write protection for RTC registers */
205  __HAL_RTC_WRITEPROTECTION_ENABLE(&hRTC_Handle);
206 
207  HAL_NVIC_SetPriority(RTC_WKUP_IRQn, TickPriority, 0U);
208  HAL_NVIC_EnableIRQ(RTC_WKUP_IRQn);
209  return HAL_OK;
210  }
211  }
212  return HAL_ERROR;
213 }
