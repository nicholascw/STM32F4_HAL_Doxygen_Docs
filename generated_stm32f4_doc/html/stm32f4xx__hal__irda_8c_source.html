767 HAL_StatusTypeDef HAL_IRDA_Transmit(IRDA_HandleTypeDef *hirda, uint8_t *pData, uint16_t Size, uint32_t Timeout)
768 {
769  uint16_t *tmp;
770  uint32_t tickstart = 0U;
771 
772  /* Check that a Tx process is not already ongoing */
773  if (hirda->gState == HAL_IRDA_STATE_READY)
774  {
775  if ((pData == NULL) || (Size == 0U))
776  {
777  return HAL_ERROR;
778  }
779 
780  /* Process Locked */
781  __HAL_LOCK(hirda);
782 
783  hirda->ErrorCode = HAL_IRDA_ERROR_NONE;
784  hirda->gState = HAL_IRDA_STATE_BUSY_TX;
785 
786  /* Init tickstart for timeout managment*/
787  tickstart = HAL_GetTick();
788 
789  hirda->TxXferSize = Size;
790  hirda->TxXferCount = Size;
791  while (hirda->TxXferCount > 0U)
792  {
793  hirda->TxXferCount--;
794  if (hirda->Init.WordLength == IRDA_WORDLENGTH_9B)
795  {
796  if (IRDA_WaitOnFlagUntilTimeout(hirda, IRDA_FLAG_TXE, RESET, tickstart, Timeout) != HAL_OK)
797  {
798  return HAL_TIMEOUT;
799  }
800  tmp = (uint16_t *) pData;
801  hirda->Instance->DR = (*tmp & (uint16_t)0x01FF);
802  if (hirda->Init.Parity == IRDA_PARITY_NONE)
803  {
804  pData += 2U;
805  }
806  else
807  {
808  pData += 1U;
809  }
810  }
811  else
812  {
813  if (IRDA_WaitOnFlagUntilTimeout(hirda, IRDA_FLAG_TXE, RESET, tickstart, Timeout) != HAL_OK)
814  {
815  return HAL_TIMEOUT;
816  }
817  hirda->Instance->DR = (*pData++ & (uint8_t)0xFF);
818  }
819  }
820 
821  if (IRDA_WaitOnFlagUntilTimeout(hirda, IRDA_FLAG_TC, RESET, tickstart, Timeout) != HAL_OK)
822  {
823  return HAL_TIMEOUT;
824  }
825 
826  /* At end of Tx process, restore hirda->gState to Ready */
827  hirda->gState = HAL_IRDA_STATE_READY;
828 
829  /* Process Unlocked */
830  __HAL_UNLOCK(hirda);
831 
832  return HAL_OK;
833  }
834  else
835  {
836  return HAL_BUSY;
837  }

1706 void HAL_IRDA_IRQHandler(IRDA_HandleTypeDef *hirda)
1707 {
1708  uint32_t isrflags = READ_REG(hirda->Instance->SR);
1709  uint32_t cr1its = READ_REG(hirda->Instance->CR1);
1710  uint32_t cr3its = READ_REG(hirda->Instance->CR3);
1711  uint32_t errorflags = 0x00U;
1712  uint32_t dmarequest = 0x00U;
1713 
1714  /* If no error occurs */
1715  errorflags = (isrflags & (uint32_t)(USART_SR_PE | USART_SR_FE | USART_SR_ORE | USART_SR_NE));
1716  if (errorflags == RESET)
1717  {
1718  /* IRDA in mode Receiver -----------------------------------------------*/
1719  if (((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
1720  {
1721  IRDA_Receive_IT(hirda);
1722  return;
1723  }
1724  }
1725 
1726  /* If some errors occur */
1727  if ((errorflags != RESET) && (((cr3its & USART_CR3_EIE) != RESET) || ((cr1its & (USART_CR1_RXNEIE | USART_CR1_PEIE)) != RESET)))
1728  {
1729  /* IRDA parity error interrupt occurred -------------------------------*/
1730  if (((isrflags & USART_SR_PE) != RESET) && ((cr1its & USART_CR1_PEIE) != RESET))
1731  {
1732  hirda->ErrorCode |= HAL_IRDA_ERROR_PE;
1733  }
1734 
1735  /* IRDA noise error interrupt occurred --------------------------------*/
1736  if (((isrflags & USART_SR_NE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
1737  {
1738  hirda->ErrorCode |= HAL_IRDA_ERROR_NE;
1739  }
1740 
1741  /* IRDA frame error interrupt occurred --------------------------------*/
1742  if (((isrflags & USART_SR_FE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
1743  {
1744  hirda->ErrorCode |= HAL_IRDA_ERROR_FE;
1745  }
1746 
1747  /* IRDA Over-Run interrupt occurred -----------------------------------*/
1748  if (((isrflags & USART_SR_ORE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
1749  {
1750  hirda->ErrorCode |= HAL_IRDA_ERROR_ORE;
1751  }
1752  /* Call IRDA Error Call back function if need be -----------------------*/
1753  if (hirda->ErrorCode != HAL_IRDA_ERROR_NONE)
1754  {
1755  /* IRDA in mode Receiver ---------------------------------------------*/
1756  if (((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
1757  {
1758  IRDA_Receive_IT(hirda);
1759  }
1760 
1761  /* If Overrun error occurs, or if any error occurs in DMA mode reception,
1762  consider error as blocking */
1763  dmarequest = HAL_IS_BIT_SET(hirda->Instance->CR3, USART_CR3_DMAR);
1764  if (((hirda->ErrorCode & HAL_IRDA_ERROR_ORE) != RESET) || dmarequest)
1765  {
1766  /* Blocking error : transfer is aborted
